# function to get filetype using file and cut
function get_file_type () {
  file $1 | cut -d " " -f 2
}
# function to match filetypes to correct command
function decompress () {
  FILETYPE=$(get_file_type $1)
  echo "[+] $1 matched to $FILETYPE filetype"
